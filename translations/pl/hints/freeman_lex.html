Proszę kontynuować rozwiązanie z podpowiedzi do poprzedniej misji z serii serii. 

def is_acceptable_password(password: str) -> bool:

    cond1 = len(password)> 6
    cond2 = any(char.isdigit() for char in password)
    cond3 = any(char.isalpha() for char in password) 

    return all([cond1, cond2, cond3])

Więc jak powinno wyglądać moje ostateczne rozwiązanie dla tej misji?

Oto ono: 

def is_acceptable_password(password: str) -> bool:

    cond1 = len(password)> 6
    cond2 = any(char.isdigit() for char in password)
    cond3 = any(char.isalpha() for char in password)
    if len(password)> 9: # cond4    
        cond2 = cond3 = True

    return all([cond1, cond2, cond3])

def is_acceptable_password(x: str) -> bool:

    return (len(x)>6 and not x.isdigit() and not x.isalpha()) or len(x)>9
